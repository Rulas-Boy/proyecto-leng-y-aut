class Prueba <|
    main(String[] Args) <|
        ~Operaciones op;

        op.@Mayor(5, 3);
        Console.@print(op.Usos);
    |>

    class [private] Operaciones <|
        int [private] Usos : 0;

        int Mayor (int x, int y) <|
            @IncrementoCont;
            
            return (x > y) ? x # y;
        |>

        void IncrementoCont <|
            Usos++2;
            2++Usos;
        |>
    |>
|>
