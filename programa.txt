main(String[] Args) <|
    ~Operaciones op;

    op.@Mayor(5, 3);
    Console.@print(op.Usos);
|>

class [private] Operaciones <|
    int [private] Usos : 0;

    int Mayor (int x, int y) <|
        bool Resultado;

        try <|
            Resultado : (x > y) ? x # y;
        |>
        catch (Exception e) <|
            Console.@print(op.Usos);
        |>

        @IncrementoCont;

        return Resultado;
    |>

    void IncrementoCont <|
        Usos++2;
        2++Usos;
    |>
|>
